void handleRoot() {
  long duration, distance;
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  String s3 = "Distance: " + String(distance) + " cm"; // Update s3 string with distance value
  tft.setGFXFont(&FreeSans12pt7b);  // Change font
  int16_t x, y, w, h;
  tft.getGFXTextExtent(s3, x, y, &w, &h); // Get string extents
  y = tft.height() - h; // Set y position to bottom of screen minus string height
  tft.fillRect(0, y, tft.width(), h, COLOR_BLACK); // Clear previous string by drawing a black rectangle
  tft.drawGFXText(x, y, s3, COLOR_GREEN); // Print updated string
}
