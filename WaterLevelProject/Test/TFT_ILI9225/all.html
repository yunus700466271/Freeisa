The LCD TFT ili9325 is a type of thin-film transistor (TFT) liquid crystal display (LCD) controller that is commonly used to drive color LCD displays in embedded systems. It is manufactured by ILI Technology Corp. and is widely used in various electronic devices, such as smartphones, digital cameras, and portable media players.

It also includes a built-in gamma correction circuit for improved image quality, as well as a touch screen controller that can support resistive and capacitive touch screens.

Overall, the ili9325 is a versatile and widely-used LCD controller that offers a range of features and capabilities for driving color displays in embedded systems.

  server.send(200, "text/html", html);
}

const int echoPin = 15;

// Define pin connections for TFT_ILI9225 display
const int TFT_CS = 5;
const int TFT_RST = 17;
const int TFT_DC = 16;

// Initialize display

// Function to measure distance
float getDistance() {
  // Send ultrasonic pulse
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
 
  // Read time for sound wave to return
  float duration = pulseIn(echoPin, HIGH);
 
  // Calculate distance in cm
  float distance = duration * 0.034 / 2;
 
  return distance;
}

  // Start serial communication for debugging
  Serial.begin(115200);
 
  // Set pin modes for JSN-SR04T sensor
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  WiFi.softAP(ssid, password);
 
  // Start web server
  server.begin();
 
  // Initialize display
  tft.begin();
  tft.setRotation(3);
}

  // Check for incoming client requests
  WiFiClient client = server.available();
 
  if (client) {
    // Read distance from sensor
    float distance = getDistance();
   
    // Convert distance to string
    String distanceString = String(distance, 2);
   
    // Build HTML page with distance value
    String html = "<html><head><title>Distance Sensor</title>";

    client.println("Content-Type: text/html");
    client.println("Connection: close");

    client.println(html);
   
    // Wait a bit for the client to receive the response

  // Clear display and write distance value to screen
